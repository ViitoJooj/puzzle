package builders

import (
	"context"
	"database/sql"
)

type BaseBuilder struct {
	db    *sql.DB
	where []string
	args  []interface{}
}

func (b *BaseBuilder) Where(cond string, args ...interface{}) {
	b.where = append(b.where, cond)
	b.args = append(b.args, args...)
}

func (b *BaseBuilder) Exec(ctx context.Context, sql string, args ...interface{}) (sql.Result, error) {
	return b.db.ExecContext(ctx, sql, args...)
}
